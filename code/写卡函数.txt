//判断是否为蓝焰IC卡
function IsLYCard(port:byte;br:integer;var EDes:pchar):Integer;stdcall;
var
  sid,strE:string;
  iyr:integer;
begin
  if icdev<=0 then
    icdev:=auto_init(port,br);
  if icdev<0 then
  begin
    EDes:=PChar('写卡器未连接');
    Result:=1;
    Exit;
  end;
  if not ReadCardID(sid,strE) then
  begin
    EDes:=pchar(strE);
    Result:=ecode;
    Exit;
  end;
  //判断sid是否为蓝焰格式
  {if sid='FFFFFFFFFFFF' then
  begin
    EDes:=PChar('非蓝焰IC卡');
    Result:=14;
    Exit;
  end;        }
  if not TryStrToInt(Copy(sid,3,2),iyr) then
  begin
    EDes:=PChar('非蓝焰IC卡');
    Result:=14;
    Exit;
  end;
  if (iyr<0) or (iyr>99) then
  begin
    EDes:=PChar('非蓝焰IC卡');
    Result:=14;
    Exit;
  end;
  Result:=0;
end;

unit unitMWRW;

interface
uses SysUtils,StrUtils,unitMWIC;

  function WriteUserCard(bh:integer;ql:Double;bqcs:Byte;srzl:integer;
            psw:string;mmcd:Boolean;out ErrorDes:string): Boolean;
  function ReadCardID(var id:string;out ErrorDes:string):Boolean;          
  function ReadUserCard(var id:string;var ickh:Integer; var gasvalue:double;
            var fc:Byte;var tpsw:boolean;out ErrorDes:string):Boolean;
  function InitICCard(psw:string;out ErrorDes:string):Boolean;
  function CheckICPswd(psw:string;out ErrorDes:string):Boolean;
  function CheckNewCard(out ErrorDes:string):Boolean;
var
  icdev:longint;
  ecode:Integer;

implementation

function WriteUserCard(bh:integer;ql:Double;bqcs:Byte;srzl:integer;
            psw:string;mmcd:Boolean;out ErrorDes:string): Boolean;
var
  status,st:integer;
  password:array[0..3] of Char;
  wbuff:TBuff;
  //iQl:Integer;
  strCardStyle: string;
  strTmp:string;
  ec:array[0..1] of Char;
begin
  Result:=False;
  //检测读写器中是否有卡
  status:=0;
  st:=get_status(icdev,@status);
  if st=0 then
  begin
    if status=0 then
    begin
      ErrorDes:='IC卡未插入';
      ecode:=3;
      dv_beep(icdev,1);
      Exit;
    end;
  end else
  begin
    ErrorDes:='写卡器状态错误';
    ecode:=2;
    Exit;
  end;
  //检测是否4442卡
  st:=chk_4442(icdev);
  if st<>0 then
  begin
    ErrorDes:='卡类型不正确';
    ecode:=4;
    dv_beep(icdev,1);
    Exit;
  end;
  //读密码错误计数，为0则表示卡已无法使用
  st:=rsct_4442(icdev,ec);
  if st<0 then
  begin
    ErrorDes:='读密码错误计数失败';
    ecode:=6;
    Exit;
  end;
  if ec[0]=chr(0) then
  begin
    ErrorDes:='该IC卡已无法继续使用';
    ecode:=7;
    Exit;
  end;
  //核对IC卡密码
  st:=turn_off(icdev);
  if st<>0 then
  begin
    ErrorDes:='对卡下电失败';
    ecode:=15;
    Exit;
  end;
  st:=turn_on(icdev);
  if st<>0 then
  begin
    ErrorDes:='对卡上电失败';
    ecode:=16;
    Exit;
  end;
  if mmcd=false then            //非密码传递卡
  begin
    asc_hex(pchar(psw),password,3);
    st:=csc_4442(icdev,3,password);
    if st<>0 then
    begin
      ErrorDes:='IC卡密码核对错误';
      ecode:=8;
      Exit;
    end;
  end else                     //密码传递卡
  begin
    //asc_hex(pchar('b62307'),password,3);
    password[0]:=chr($B6);
    password[1]:=chr($23);
    password[2]:=chr($07);
    st:=csc_4442(icdev,3,password);
    if st<>0 then
    begin
      ErrorDes:='IC卡密码核对错误';
      ecode:= 8;
      Exit;
    end;
    //此处需要修改IC卡密码
    asc_hex(pchar(psw),password,3);
    st:=wsc_4442(icdev,3,password);
    if st<>0 then
    begin
      ErrorDes:='修改IC卡密码错误';
      ecode:=9;
      Exit;
    end;
  end;
  //判断是否为工业表
  if not ReadCardID(strCardStyle,ErrorDes) then
    Exit;
  strCardStyle:=Copy(strCardStyle,7,1);
  //写卡........................
  wbuff[0]:=chr($DD);
  //ic卡编号
  wbuff[1]:=chr(bh div 1000000);
  wbuff[2]:=chr(bh div 10000 mod 100);
  wbuff[3]:=chr(bh div 100 mod 100);
  wbuff[4]:=chr(bh mod 100);
  //密码区
  if mmcd=true then
  begin
    wbuff[5]:=chr(StrToInt('$'+Copy(psw,1,2)));
    wbuff[6]:=chr(StrToInt('$'+Copy(psw,3,2)));
    wbuff[7]:=chr(StrToInt('$'+Copy(psw,5,2)));
  end else
  begin
    wbuff[5]:=chr(0);
    wbuff[6]:=chr(0);
    wbuff[7]:=chr(0);
  end;
  //气量
  strTmp:=Trim(format('%5.1f',[ql]));
  strTmp:=StringOfChar('0',7-Length(strTmp))+strTmp;
  if strCardStyle='1' then
  begin
    wbuff[8]:=chr(StrToInt(Copy(strTmp,1,2)));
    wbuff[9]:=chr(StrToInt(Copy(strTmp,3,2)));
    wbuff[10]:=chr(StrToInt(Copy(strTmp,5,1))*10+StrToInt(Copy(strTmp,7,1)));
  end else
  begin
    wbuff[8]:=chr(StrToInt(Copy(strTmp,3,1)));
    wbuff[9]:=chr(StrToInt(Copy(strTmp,4,2)));
    wbuff[10]:=chr(StrToInt(Copy(strTmp,7,1)));
  end;
  //未用
  wbuff[11]:=chr(0);
  wbuff[12]:=chr(0);
  wbuff[13]:=chr(0);
  //输入总量区
  strTmp:=Trim(format('%6d',[srzl]));
  strTmp:=StringOfChar('0',6-Length(strTmp))+strTmp;
  if strCardStyle='1' then
  begin
    wbuff[14]:=chr(StrToInt(Copy(strTmp,1,1)));
    wbuff[15]:=chr(StrToInt(Copy(strTmp,2,2)));
    wbuff[16]:=chr(StrToInt(Copy(strTmp,4,2)));
  end else
  begin
    wbuff[14]:=chr(StrToInt(Copy(strTmp,1,2)));
    wbuff[15]:=chr(StrToInt(Copy(strTmp,3,2)));
    wbuff[16]:=chr(StrToInt(Copy(strTmp,5,2)));
    end;
  //未用
  wbuff[17]:=chr(0);
  //密码传递标志
  if mmcd=true then
    wbuff[18]:=chr($AA)
  else
    wbuff[18]:=chr(0);
  //补气次数
  wbuff[19]:=Chr(bqcs);
  st:=swr_4442(icdev,32,20,wbuff);
  if st<0 then
  begin
    ErrorDes:='写卡异常';
    ecode:=10;
    Exit;
  end;
  dv_beep(icdev,20);
  Result:=True;
end;

function ReadCardID(var id:string;out ErrorDes:string):Boolean;
var
  rbuff:TBuff;
  st,status:Integer;
  str:String;
  i:Integer;
begin
  Result:=False;
  //检测读写器中是否有卡
  status:=0;
  st:=get_status(icdev,@status);
  if st=0 then
  begin
    if status=0 then
    begin
      ErrorDes:='IC卡未插入';
      ecode:=3;
      dv_beep(icdev,1);
      Exit;
    end;
  end else
  begin
    ErrorDes:='写卡器状态错误';
    ecode:=2;
    Exit;
  end;
  //检测是否4442卡
  st:=chk_4442(icdev);
  if st<>0 then
  begin
    ErrorDes:='卡类型不正确';
    ecode:=4;
    dv_beep(icdev,1);
    Exit;
  end;
  //读唯一标示号
  st:=srd_4442(icdev,10,6,rbuff);
  if st<>0 then
  begin
    ErrorDes:='读卡异常';
    ecode:=11;
    Exit;
  end;
  id:='';
  for i:=0 to 5 do
  begin
    str:=Trim(format('%2x',[Ord(rbuff[i])]));
    str:=StringOfChar('0',2-(Length(str)))+str;
    id:=id+str;
  end;
  Result:=True;
end;

function ReadUserCard(var id:string;var ickh:Integer; var gasvalue:double;
            var fc:Byte;var tpsw:boolean;out ErrorDes:string):Boolean;
var
  rbuff:TBuff;
  st,status:Integer;
  str:String;
  i:Integer;
  strCardStyle: string;
begin
  Result:=False;
  //检测读写器中是否有卡
  status:=0;
  st:=get_status(icdev,@status);
  if st=0 then
  begin
    if status=0 then
    begin
      ErrorDes:='IC卡未插入';
      ecode:=3;
      dv_beep(icdev,1);
      Exit;
    end;
  end else
  begin
    ErrorDes:='写卡器状态错误';
    ecode:=2;
    Exit;
  end;
  //检测是否4442卡
  st:=chk_4442(icdev);
  if st<>0 then
  begin
    ErrorDes:='卡类型不正确';
    ecode:=4;
    dv_beep(icdev,1);
    Exit;
  end;
  //读唯一标示号
  st:=srd_4442(icdev,10,6,rbuff);
  if st<>0 then
  begin
    ErrorDes:='读卡异常';
    ecode:=11;
    Exit;
  end;
  id:='';
  for i:=0 to 5 do
  begin
    str:=Trim(format('%2x',[Ord(rbuff[i])]));
    str:=StringOfChar('0',2-(Length(str)))+str;
    id:=id+str;
  end;
  strCardStyle:=Copy(id,7,1);
  //读卡上用户数据
  st:=srd_4442(icdev,32,20,rbuff);
  if st<>0 then
  begin
    ErrorDes:='读卡异常';
    ecode:=11;
    Exit;
  end;
  if rbuff[0]<>Chr($DD) then
  begin
    ErrorDes:='非用户卡';
    ecode:=5;
    Exit;
  end;
  ickh:=Ord(rbuff[1])*1000000+Ord(rbuff[2])*10000+Ord(rbuff[3])*100+Ord(rbuff[4]);
  if strCardStyle='0'then
    gasvalue:=Ord(rbuff[8])*100+Ord(rbuff[9])+Ord(rbuff[10])/10;
  if strCardStyle='1' then
    gasvalue:=Ord(rbuff[8])*1000+Ord(rbuff[9])*10+Ord(rbuff[10])/10;
  fc:=Ord(rbuff[19]);
  if rbuff[18]=chr($AA) then
    tpsw:=True
  else
    tpsw:=False;
  dv_beep(icdev,20);  
  Result:=true;    
end;

function CheckNewCard(out ErrorDes:string):Boolean;
var
  st:integer;
  rbuff:TBuff;
  iYwbh:integer;
begin
  Result:=False;
  st:=srd_4442(icdev,32,20,rbuff);
  if st<>0 then
  begin
    ErrorDes:='读卡异常';
    ecode:=11;
    Exit;
  end;
  if rbuff[0]<>Chr($DD) then
  begin
    ErrorDes:='非用户卡';
    ecode:= 5;
    Exit;
  end;
  iYwbh:=Ord(rbuff[1])*1000000+Ord(rbuff[2])*10000+Ord(rbuff[3])*100+Ord(rbuff[4]);
  if iYwbh<>0 then
  begin
    ErrorDes:='IC卡用户编号存在';
    ecode:=12;
    Exit;
  end;
  Result:=CheckICPswd('b62307',ErrorDes);
end;
